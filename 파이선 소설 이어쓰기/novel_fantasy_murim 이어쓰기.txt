# 괄호 안에 대공장주님의 '진짜 유료 키'를 작은따옴표 안에 정확히 넣으시면 됩니다.
client = genai.Client(api_key='여기에_진짜_유료_키_직접_넣기')
MODEL_NAME = 'gemini-2.5-flash'

def generate_fusion(prompt, description="요청"):
    """구글 가드레일 및 빈 값(None) 반환 에러를 완벽하게 방어하는 안전 업그레이드 함수"""
    for attempt in range(1, 4):  # 최대 3번까지 자동 재시도
        try:
            response = client.models.generate_content(model=MODEL_NAME, contents=prompt)
            if response and response.text and response.text.strip():
                return response.text
            print(f"\n⚠️ [가드레일 감지] 구글 필터링으로 인해 빈 내용이 반환되었습니다. {attempt}차 우회 재시도 중...")
            time.sleep(2)
        except Exception as e:
            print(f"\n🔄 [무림북 서버 지연] {description} {attempt}차 재시도 중... (오류 발생으로 대기)")
            time.sleep(3)
    return "심장의 마나 서클과 단전의 기경팔맥이 동시에 공명하기 시작했다. 차원을 초월한 절대자의 질주가 시작된다."

print("==================================================")

        synopses[i] = f"{novel_title}의 흥미진진한 제 {i}화 스토리가 전개됩니다."

# 📝 [무협 판타지 작가 규칙] 무협의 기운과 판타지 마법을 융합하기 위한 규칙
# ======================================================================
HYBRID_RULES = """
당신은 대한민국 최고의 스타 퓨전 소설 작가입니다. 다음 금지 및 권장 사항을 엄격히 준수하여 집필하세요.

[절대 금지 사항]
1. 지문 끝에 (~지, ~으니, ~네, ~소) 같은 낡은 어조는 금지합니다. 세련되고 몰입감 높은 현대 웹소설 문법(~했다, ~였다)으로 서술하십시오.
2. 본문 어디에도 한자(漢字) 자체 표기 및 괄호 병기를 절대 금지합니다. 오직 순수한 '한글'로만 표기하십시오.
3. 소설 본문이 아닌 다른 잡설은 절대 출력하지 마십시오. 오직 소설 본문만 나와야 합니다.

[권장 스타일 및 차원 융합 도파민 요소]
- 중원의 무공(내공, 기경팔맥, 삼류, 화산파, 매화검기, 강기 등)과 서양 판타지(마나 서클, 파이어볼, 드래곤, 서양 검술, 서양 귀족, 아카데미 등)가 격돌하거나 융합하는 전투 묘사를 최소 3문장 이상 아주 웅장하게 집필하십시오.
- 서양식 오라를 쓰는 기사단장이나 8서클 마법사가 주인공이 펼치는 무협식 '검강'이나 '신법'을 보고 "저 기괴하고 웅장한 기운의 정체는 뭐지?!"라며 입을 쩍 벌리는 '서양인 경악 사이다 리액션'을 꼭 포함하십시오.
- 문장은 빠른 가독성을 위해 적당히 호흡을 끊어 줄바꿈해 주시고, 대사 비중을 높여 사이다 전개를 이어가세요.
"""

previous_summary = "없음 (기존 회차로부터 연결됩니다.)"

info_filename = os.path.join(output_dir, f"00_🚨작품소개_및_기획안🚨.txt")

    with open(info_filename, "w", encoding="utf-8") as info_f:

            previous_summary = generate_fusion(summary_prompt, f"{chapter}화 기존 줄거리 요약")

        continue

    prompt = f"""

    [소설 기본 정보] - 제목: {novel_title}

    """
    
    chapter_text = generate_fusion(prompt, f"{chapter:02d}화 본문 집필")

    summary_prompt = f"다음 소설 본문을 읽고 주인공의 차원 초월 성장 상태나 주변인들의 경악 반응을 3줄 요약해줘.\n\n[본문]\n{chapter_text}"

print(f"🎉 [공장 가동 완료] 소설 <{novel_title}>이 완벽하게 50화 완결되었습니다!")
print(f"📂 [{output_dir}] 폴더에서 차원을 초월한 퓨전 신작을 확인하세요!")
