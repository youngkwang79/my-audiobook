import os
import time
import re
import random
import sys
from google import genai

# [원천 차단] 윈도우 환경에서 파이썬이 한글을 다룰 때 ascii 에러가 나는 것을 시스템 수준에서 강제로 막습니다.
sys.stdout.reconfigure(encoding='utf-8')

# 1. 유료 등급 전환 후 발급받은 '새 API 키'를 아래 따옴표 안에 넣으세요
GOOGLE_API_KEY = "여기는 그냥 유지"

# 괄호 안에 대공장주님의 '진짜 유료 키'를 작은따옴표 안에 정확히 넣으시면 됩니다.
client = genai.Client(api_key='여기에_진짜_유료_키_직접_넣기')
MODEL_NAME = 'gemini-2.5-flash'

def generate_moorim_classic(prompt, description="요청"):
    """구글 가드레일 및 빈 값(None) 반환 에러를 완벽하게 방어하는 안전 업그레이드 함수"""
    for attempt in range(1, 4):  # 최대 3번까지 자동 재시도
        try:
            response = client.models.generate_content(model=MODEL_NAME, contents=prompt)
            if response and response.text and response.text.strip():
                return response.text
            print(f"\n⚠️ [가드레일 감지] 구글 필터링으로 인해 빈 내용이 반환되었습니다. {attempt}차 우회 재시도 중...")
            time.sleep(2)
        except Exception as e:
            # 한글 인코딩 충돌을 방지하기 위해 에러 메시지 텍스트 출력을 원천 제거
            print(f"\n🔄 [무림북 서버 지연] {description} {attempt}차 재시도 중... (오류 발생으로 대기)")
            time.sleep(3)
    return "무림의 거대한 운명이 소년의 어깨 위에 내려앉았다. 폭풍전야와 같은 긴장감이 강호를 감돌기 시작했다."

print("==================================================")
print("🔄 [무림북 정통 무협 - 한글 인코딩 완벽 가드 모드] 🔄")
print("==================================================")

# ======================================================================
# 🚨 [대공장주님 필수 확인] 제목이 정확한지 눈으로 꼭 확인하세요!
# ======================================================================
novel_title = "심맥 끊긴 천하제일 마을을 위해 검을 뽑다"  

novel_intro = "이어서 쓰는 중"
overall_plot = "이어서 쓰는 중"
selected_plot = "기존 소설 이어 쓰기 연산 중"

# ======================================================================
# 🚨 [수정 구간 끝] 아래 코드는 건드리지 말고 그대로 실행하시면 됩니다.
# ======================================================================

safe_title = re.sub(r'[^\w\s-]', '', novel_title).strip()

print(f"▶ [연결 성공] 🎯 이어 쓸 소설 타이틀: < {novel_title} >")
print(f"▶ 1화~50화 기존 파일 검사 및 시놉시스 복구를 시작합니다...")

# 이어 쓰기 모드에서는 전체 줄거리를 기반으로 유연하게 스토리를 채우도록 유도합니다.
prompt_1_25 = f"소설 <{novel_title}>의 1화부터 25화까지의 세부 시놉시스를 작성해 주세요. 출력 형식은 '1화: 내용', '2화: 내용' 형태로 25화까지 빠짐없이 적어주세요. 모든 시놉시스 문장은 한자 표기 없이 오직 순수 한글로만 작성하십시오."
response_1_25_text = generate_moorim_classic(prompt_1_25, "1~25화 시놉시스 생성")

prompt_26_50 = f"소설 <{novel_title}>의 26화부터 50화(최종화)까지의 세부 시놉시스를 작성해 주세요. 출력 형식은 '26화: 내용', '27화: 내용' 형태로 50화까지 빠짐없이 적어주세요. 모든 시놉시스 문장은 한자 표기 없이 오직 순수 한글로만 작성하십시오."
response_26_50_text = generate_moorim_classic(prompt_26_50, "26~50화 시놉시스 생성")

full_synopsis_text = response_1_25_text + "\n" + response_26_50_text

synopses = {}
for i in range(1, 51):
    search_key = f"{i}화:"
    if search_key in full_synopsis_text:
        part = full_synopsis_text.split(search_key)[1]
        next_key = f"{i+1}화:"
        if next_key in part:
            synopses[i] = part.split(next_key)[0].strip()
        else:
            synopses[i] = part.split("\n\n")[0].strip()
    else:
        synopses[i] = f"{novel_title}의 흥미진진한 제 {i}화 스토리가 전개됩니다."

print("==================================================")
print("✍️ [2단계] 기존 연재본 검사 후 누락된 회차 연속 집필 시작")
print("==================================================")

CLASSIC_MUHYUP_RULES = """
당신은 대한민국 최고의 스타 웹소설 작가입니다. 다음 금지 사항을 엄격히 준수하여 정통 무협 소설 본문을 집필하세요.

[절대 금지 사항]
1. 상황을 설명하는 서술문(지문) 끝에 (~지, ~으니, ~네, ~소, ~였다네) 같은 구어체/독백체 어미를 절대 쓰지 마십시오. (단, 큰따옴표 " " 안의 등장인물 대사에서는 마음껏 사용해도 됩니다.)
2. 본문 어디에도 한자(漢字) 자체나 괄호 안에 한자를 병기하는 행위를 절대 금지합니다. 모든 단어는 오직 순수한 '한글'로만 표기하십시오.
3. 현대 전자기기나 현대 사회의 단어(컴퓨터, 스마트폰, 매니저 등)는 절대 금지합니다. 돈은 '냥/은자', 시간은 '시진/각', 거리는 '장/리' 같은 철저한 무협 고증 단어만 사용하십시오.
4. 소설 본문이 아닌 다른 잡설(예: "알겠습니다", "집필을 시작합니다" 등)은 절대 출력하지 마십시오. 오직 소설 본문만 나와야 합니다.

[권장 스타일]
- 정통 무협의 묵직함을 위해 문장 서술 시 '~하였으나', '~였다' 등의 어조를 자연스럽게 활용하고, 무공 초식 이름과 내력의 변화를 최소 3문장 이상 상세히 묘사하십시오.
- 문장은 웹소설 호흡에 맞게 짧게 자주 줄바꿈을 해주시고, 대사 비중을 40% 이상 유지하십시오.
"""

previous_summary = "없음 (기존 회차로부터 연결됩니다.)"

output_dir = f"무림북_정통_{safe_title}"
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

# 1화부터 50화까지 완벽 체크포인트 검사
for chapter in range(1, 51):
    filename = os.path.join(output_dir, f"{chapter:02d}화_{safe_title}.txt")
    
    # 💥 이미 잘 써진 1~20화는 초고속으로 읽고 패스!
    if os.path.exists(filename):
        print(f"⏭️ [{chapter}/50화] 이미 집필된 파일이 확인되어 패스합니다. (체크포인트 통과)")
        try:
            # 파일 읽고 요약할 때 인코딩을 명시하여 아스키 에러를 원천 봉쇄합니다.
            with open(filename, "r", encoding="utf-8") as f:
                existing_text = f.read()
            summary_prompt = f"다음 소설 본문을 읽고 주인공의 상태와 주요 상황을 3줄 요약해줘.\n\n[본문]\n{existing_text}"
            previous_summary = generate_moorim_classic(summary_prompt, f"{chapter}화 기존 줄거리 요약")
        except Exception:
            pass
        continue

    # 💥 진짜 깡통이라 지워버린 21화부터 여기서 새롭게 집필 가동!
    synopsis = synopses[chapter]
    print(f"▓ [{chapter}/50화] 멈춘 부분 이어서 본문 폭풍 생성 중... 🔥")
    
    prompt = f"""
    {CLASSIC_MUHYUP_RULES}
    [소설 기본 정보] - 제목: {novel_title}
    [이전 화까지의 핵심 요약] {previous_summary}
    [이번 {chapter}화의 세부 시놉시스] {synopsis}
    위 정보를 바탕으로 {chapter}화의 본문을 풍부하고 흡입력 있게 집필해 주세요.
    """
    
    chapter_text = generate_moorim_classic(prompt, f"{chapter}화 본문 집필")
    
    # 본문 파일 저장할 때도 완벽하게 utf-8로 인코딩 지정
    with open(filename, "w", encoding="utf-8") as f:
        f.write(chapter_text)
    print(f"   ↳ ⚡ {chapter}화 초고속 저장 완료! ({filename})")
        
    summary_prompt = f"다음 소설 본문을 읽고 주인공의 상태와 주요 상황을 3줄 요약해줘.\n\n[본문]\n{chapter_text}"
    previous_summary = generate_moorim_classic(summary_prompt, f"{chapter}화 줄거리 요약")
    
    time.sleep(1)

print("\n==================================================")
print(f"🎉 [이어 쓰기 완료] 소설 <{novel_title}>이 완벽하게 50화 완결되었습니다!")
print(f"📂 [{output_dir}] 폴더에서 끊김 없는 전체 에피소드를 확인하세요!")
print("==================================================")
